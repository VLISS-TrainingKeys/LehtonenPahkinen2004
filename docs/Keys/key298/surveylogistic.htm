CODE for the analysis of the model
PSYCH2 = Intercept + AGE + SEX + CHRON:

proc surveylogistic data=ohc;

strata Stratum;

cluster Cluster;

weight _one_;

model psych2(event='1')=age sex chron / link=logit rsquare;

run;
